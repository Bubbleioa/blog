【考前冲刺Day6】OI STILE

T1 引子（水箱）

非常简单的模拟题目，错误点有两处：

  1. 没有读入多位数字
  2. 出现顺序和编号无关

然就是从1号水箱，开始递归，优先从箱底的水管递归下去，然后输出自身的编号。

#include&lt;bits/stdc++.h&gt;
using namespace std;
const int MAXN=1003;
char mp[MAXN][MAXN];
int n,m;
int draw[MAXN][MAXN];
int numx1[200003],numy1[200003];
int numx2[200003],numy2[200003];
  
void paint(int col,int i,int j){
    int x=i,y=j;
    while(mp[x][y]!=&#039;|&#039;)y++;
    while(mp[x][y]!=&#039;+&#039;)x++;
    while(mp[i][j]!=&#039;|&#039;)j--;
    while(mp[i][j]!=&#039;+&#039;)i--;
    for(int k=i;k&lt;=x;k++){
        for(int z=j;z&lt;=y;z++){
            draw[k][z]=col;
        }
    }
    numx1[col]=i;numy1[col]=j;
    numx2[col]=x;numy2[col]=y;
}
  
int go(int x,int y,int lasty){
    int flag=1;
    while(1){
        if(flag==1){
            int r=y,l=y;
            if(y&gt;lasty){
                while(mp[x][r]!=&#039;+&#039;&amp;&amp;mp[x][r]==&#039;-&#039;)r++;
            }
            else {
                while(mp[x][l]!=&#039;+&#039;&amp;&amp;mp[x][l]==&#039;-&#039;)l--;
            }
            if(y&gt;lasty){
                if(mp[x][r]==&#039;+&#039;)y=r;
            }
            else {
                if(mp[x][l]==&#039;+&#039;)y=l;
            }
            x++;
            flag=2;
        }
        else {
            while(mp[x][y]!=&#039;+&#039;&amp;&amp;mp[x][y]==&#039;|&#039;)x++;
            if(draw[x][y]!=0)return draw[x][y];
            if(mp[x][y]==&#039;+&#039;){lasty=y;flag=1;if(mp[x][y+1]==&#039;-&#039;)y++;else y--;}
        }
    }
}
      
  
void dfs(int id){
    for(int i=numx2[id]-1;i&gt;=numx1[id]+1;i--){
        if(mp[i][numy1[id]-1]==&#039;-&#039;){
            dfs(go(i,numy1[id]-1,numy1[id]));
        }
        if(mp[i][numy2[id]+1]==&#039;-&#039;){
            dfs(go(i,numy2[id]+1,numy2[id]));
        }
    }
    printf(&quot;%d\n&quot;,id);
}
  
int main(){
    scanf(&quot;%d%d&quot;,&amp;n,&amp;m);
    for(int i=1;i&lt;=n;i++)scanf(&quot;%s&quot;,mp[i]+1);
    for(int i=1;i&lt;=n;i++){
        for(int j=1;j&lt;=m;j++){
            int num=0;
            while(isdigit(mp[i][j])){
                num=num*10+mp[i][j]-&#039;0&#039;;
                j++;
            }
            if(num!=0)paint(num,i,j-1);
        }
    }
    dfs(1);
    return 0;
}

T2 可爱精灵宝贝

一道区间dp题，考场上写挂了，最后10分钟乱搞居然也有60分，考试完调了一下，有90分实际上是数据太水了。

90分代码：

#include&lt;bits/stdc++.h&gt;
using namespace std;
const int MAXN=1005;
struct node{
    int a,b,t;
}go[MAXN];
int n,k,m;
int maxt;
 
int main(){
    scanf(&quot;%d%d%d&quot;,&amp;n,&amp;k,&amp;m);
    for(int i=1;i&lt;=m;i++){
        scanf(&quot;%d%d%d&quot;,&amp;go[i].a,&amp;go[i].b,&amp;go[i].t);
        maxt=max(maxt,go[i].t);
    }
    int mx=0;
    for(int i=1;i&lt;=k-1;i++){
        int now=0;
        int x=k;int ans=0;
        while(x!=i-1&amp;&amp;now&lt;=maxt){
            now++;
            for(int j=1;j&lt;=m;j++){
                if(go[j].a==x&amp;&amp;go[j].t&gt;=now)ans+=go[j].b;
            }
            x--;
        }
        if(now){
            now--;
            now*=2;    
        }
        x=k;
        while(now&lt;=maxt&amp;&amp;x&lt;=n){
            for(int j=1;j&lt;=m;j++){
                if(go[j].a==x&amp;&amp;go[j].t&gt;=now)ans+=go[j].b;
            }
            now++;
            x++;
        }
        mx=max(mx,ans);
    }
    for(int i=n;i&gt;=k;i--){
        int now=0;
        int x=k;int ans=0;
        while(x!=i+1&amp;&amp;now&lt;=maxt){
            now++;
            for(int j=1;j&lt;=m;j++){
                if(go[j].a==x&amp;&amp;go[j].t&gt;=now)ans+=go[j].b;
            }
            x++;
        }
        if(now){
            now--;
            now*=2;    
        }
        x=k;
        while(now&lt;=maxt&amp;&amp;x&gt;=1){
            for(int j=1;j&lt;=m;j++){
                if(go[j].a==x&amp;&amp;go[j].t&gt;=now)ans+=go[j].b;
            }
            now++;
            x--;
        }
        mx=max(mx,ans);
    }
    cout&lt;&lt;mx&lt;&lt;endl;
    return 0;
}

这里区间dp用的是记忆化搜索实现的，dfs中的参数含义如下：

参数名 参数含义
s 第s个精灵
cur 当前时间
sum 当前状态的分数
l 当前状态的左端点
r 当前状态的右端点

其中，l和r指的是按位置排序后精灵序号的左右端点，相当于离散化了坐标

对于状态，我们都有以下选择：

  1. 向右走去捕获右边精灵
  2. 向右走，虽然捕获不到
  3. 向左走去捕获左边精灵
  4. 向左走，虽然捕获不到

对于每种状态的遍历，我们就可以算出f[i][j]—选择第i到第j个精灵的最大得分（不一定捕获到了）
答案就是其中最大的

#include&lt;bits/stdc++.h&gt;
using namespace std;
const int MAXN=1003;
int n,k,m;
int f[1003][1003];
int ans;
struct node{
    int pos;
    int val;
    int t;
}a[MAXN];
bool book[MAXN];

bool cmp(node a,node b){
    return a.pos&lt;b.pos;
}

void dfs(int s,int cur,int sum,int l,int r){
    if(f[l][r]&gt;=sum)return;
    f[l][r]=sum;
    if(sum&gt;ans)ans=sum;
    book[s]=1;
    if(!book[r]&amp;&amp;r&lt;=m){
        if(cur+abs(a[r].pos-a[s].pos)&lt;=a[r].t){
            dfs(r,cur+abs(a[r].pos-a[s].pos),sum+a[r].val,l,r+1);
        }
        else {
            dfs(r,cur+abs(a[r].pos-a[s].pos),sum,l,r+1);
        }
    }
    if(!book[l]&amp;&amp;l){
        if(cur+abs(a[s].pos-a[l].pos)&lt;=a[l].t){
            dfs(l,cur+abs(a[l].pos-a[s].pos),sum+a[l].val,l-1,r);
        }
        else {
            dfs(l,cur+abs(a[l].pos-a[s].pos),sum,l-1,r);
        }
    }
    book[s]=0;//注意回溯
}

int main(){
    scanf(&quot;%d%d%d&quot;,&amp;n,&amp;k,&amp;m);
    for(int i=1;i&lt;=m;i++){
        int d,b,c;scanf(&quot;%d%d%d&quot;,&amp;d,&amp;b,&amp;c);
        a[i]=(node){d,b,c};
    }
    a[++m]=(node){k,0,1};//从k位置开始
    sort(a+1,a+1+m,cmp);
    memset(f,-1,sizeof(f));
    for(int i=1;i&lt;=m;i++){
        if(a[i].pos==k&amp;&amp;a[i].val==0&amp;&amp;a[i].t==1){
            dfs(i,1,0,i-1,i+1);
            break;
        }
    }
    printf(&quot;%d&quot;,ans);
    return 0;
}
