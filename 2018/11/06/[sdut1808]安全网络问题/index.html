[sdut1808]安全网络问题

题面

只能看看题面，数据是错的。

就是求一张图的所有点双连通分量，将它们内部排序再外部排序数出来，关于点双连通分量没什么好讲的，网上各种博客都写烂了。还是直接上代码吧。

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;cstring&gt; 
#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;
const int MAXN=600000;
int n;
int Nt[MAXN&lt;&lt;1],to[MAXN&lt;&lt;1],Head[MAXN],tot;
int root;
int dfn[MAXN],low[MAXN],cut[MAXN];
vector&lt;int&gt;dcc[MAXN];
int num,cnt;
int st[MAXN],top;
 
void add(int a,int b){
    Nt[++tot]=Head[a];
    to[tot]=b;
    Head[a]=tot;
}
 
void tarjan(int x){
    dfn[x]=low[x]=++num;
    st[++top]=x;
    if(x==root&amp;&amp;Head[x]==0){
        dcc[++cnt].push_back(x);
        return;
    }
    int flag=0;
    for(int i=Head[x];i;i=Nt[i]){
        int y=to[i];
        if(!dfn[y]){
            tarjan(y);
            low[x]=min(low[x],low[y]);
            if(low[y]&gt;=dfn[x]){
                flag++;
                if(x!=root||flag&gt;1)cut[x]=1;
                cnt++;
                int z;
                do{
                    z=st[top--];
                    dcc[cnt].push_back(z);
                }while(z!=y);
                dcc[cnt].push_back(x);
            }
        }
        else low[x]=min(low[x],dfn[y]);
    }
}
 
bool cmp(vector&lt;int&gt;a,vector&lt;int&gt;b){
    int lim=min(a.size(),b.size());
    for(int i=0;i&lt;lim;i++){
        if(a[i]==b[i])continue;
        return a[i]&lt;b[i];
    }
    return a.size()&lt;b.size();
}
 
int main(){
    scanf(&quot;%d&quot;,&amp;n);
    for(int i=1;i&lt;=n;i++){
        int x;scanf(&quot;%d&quot;,&amp;x);x++;
        char str[7];scanf(&quot;%s&quot;,str);
        int num=0;
        for(int j=1;j&lt;strlen(str);j++){
            if(str[j]==&#039;)&#039;)break;
            num=num*10+str[j]^48;
        }
        for(int j=1;j&lt;=num;j++){
            int y;scanf(&quot;%d&quot;,&amp;y);y++;
            add(x,y);add(y,x);
        }
    }
    for(int i=1;i&lt;=n;i++){
        if(!dfn[i])root=i,tarjan(i);
    }
    for(int i=1;i&lt;=cnt;i++)sort(dcc[i].begin(),dcc[i].end());
    sort(dcc+1,dcc+1+cnt,cmp);
    cout&lt;&lt;cnt&lt;&lt;endl;
    for(int i=1;i&lt;=cnt;i++){
        for(int j=0;j&lt;dcc[i].size();j++){
            printf(&quot;%d &quot;,dcc[i][j]-1);
        }
        printf(&quot;\n&quot;);
    }
    return 0;
}
