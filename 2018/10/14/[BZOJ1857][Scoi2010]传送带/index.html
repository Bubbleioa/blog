[BZOJ1857][Scoi2010]传送带

Description

在一个2维平面上有两条传送带，每一条传送带可以看成是一条线段。两条传送带分别为线段AB和线段CD。Lxp在AB上的移动速度为P，在CD上的移动速度为Q，在平面上的移动速度R。现在Lxp想从A点走到D点，他想知道最少需要走多长时间

Input

输入数据第一行是4个整数，表示A和B的坐标，分别为Ax，Ay，Bx，By 第二行是4个整数，表示C和D的坐标，分别为Cx，Cy，Dx，Dy 第三行是3个整数，分别是P，Q，R

Output

输出数据为一行，表示Lxp从A点走到D点的最短时间，保留到小数点后2位

Sample Input

0 0 0 100
100 0 100 100
2 2 1

Sample Output

136.60

HINT

对于100%的数据，1<= Ax，Ay，Bx，By，Cx，Cy，Dx，Dy<=1000
1<=P，Q，R<=10

题解

代码

#include&lt;iostream&gt;
#include&lt;cstdio&gt;
#include&lt;cstdlib&gt;
#include&lt;ctime&gt;
#include&lt;cmath&gt;
using namespace std;
const double R=0.98,Exp=1e-3,times=50;

inline double dis(double X1,double Y1,double X2,double Y2){
    return sqrt((X1-X2)*(X1-X2)+(Y1-Y2)*(Y1-Y2));
}

int main(){
    srand((unsigned)time(0));
    double ax,ay,bx,by,cx,cy,dx,dy;
    int p,q,r;
    scanf(&quot;%lf%lf%lf%lf&quot;,&amp;ax,&amp;ay,&amp;bx,&amp;by);
    scanf(&quot;%lf%lf%lf%lf&quot;,&amp;cx,&amp;cy,&amp;dx,&amp;dy);
    scanf(&quot;%d%d%d&quot;,&amp;p,&amp;q,&amp;r);
    double ab,cd,ac,bc,ad,bd;
    ab=dis(ax,ay,bx,by);
    cd=dis(cx,cy,dx,dy);
    ac=dis(ax,ay,cx,cy);
    bc=dis(bx,by,cx,cy);
    ad=dis(ax,ay,dx,dy);
    bd=dis(bx,by,dx,dy);
    double ans=min(ad/r,min(ab/p+bd/r,min(ab/p+bc/r+cd/q,ac/r+cd/q)));
    for(int i=1;i&lt;=times;i++){
        double tx=ax,ty=ay,px=bx,py=by,mx,my,T=20070831;
        while(dis(tx,ty,px,py)&gt;Exp){//走AB，中途直接前往D
            mx=(tx+px)/2,my=(ty+py)/2;
            double fu=dis(mx,my,ax,ay)/p+dis(mx,my,dx,dy)/r;
            if(fu&lt;ans){
                ans=fu;
                if(rand()%2) tx=mx,ty=my;
                else px=mx,py=my;
            }
            else {
                if(rand()&lt;T) {
                    if(rand()%2)tx=mx,ty=my;
                    else px=mx,py=my;
                    T*=R;
                }
            }
        }
        tx=cx,ty=cy,px=dx,py=dy,mx,my,T=20070831;
        while(dis(tx,ty,px,py)&gt;Exp){//A直接走到CD某点上
            mx=(tx+px)/2,my=(ty+py)/2;
            double fu=dis(mx,my,ax,ay)/r+dis(mx,my,dx,dy)/q;
            if(fu&lt;ans){
                ans=fu;
                if(rand()%2) tx=mx,ty=my;
                else px=mx,py=my;
            }
            else {
                if(rand()&lt;T) {
                    if(rand()%2)tx=mx,ty=my;
                    else px=mx,py=my;
                    T*=R;
                }
            }
        }
        tx=ax,ty=ay,px=bx,py=by,mx,my,T=20070831;
        while(dis(tx,ty,px,py)&gt;Exp){//走AB，中途前往CD
            mx=(tx+px)/2,my=(ty+py)/2;
            double fu=dis(mx,my,ax,ay)/p+dis(mx,my,cx,cy)/r+cd/q;
            if(fu&lt;ans){
                ans=fu;
                if(rand()%2) tx=mx,ty=my;
                else px=mx,py=my;
            }
            else {
                if(rand()&lt;T) {
                    if(rand()%2)tx=mx,ty=my;
                    else px=mx,py=my;
                    T*=R;
                }
            }
        }
        tx=cx,ty=cy,px=dx,py=dy,mx,my,T=20070831;
        while(dis(tx,ty,px,py)&gt;Exp){//走完AB再前往CD
            mx=(tx+px)/2,my=(ty+py)/2;
            double fu=dis(mx,my,bx,by)/r+dis(mx,my,dx,dy)/q+ab/p;
            if(fu&lt;ans){
                ans=fu;
                if(rand()%2) tx=mx,ty=my;
                else px=mx,py=my;
            }
            else {
                if(rand()&lt;T) {
                    if(rand()%2)tx=mx,ty=my;
                    else px=mx,py=my;
                    T*=R;
                }
            }
        }
    }
    printf(&quot;%.2lf&quot;,ans);
    return 0;
}
