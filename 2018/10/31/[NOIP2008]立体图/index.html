#include&lt;bits/stdc++.h&gt;
using namespace std;
int n,m,maxx,maxy;
int mana[1000][1000];
int pic[1000][1000];

void print(){
    for(register int i=maxx;i&gt;=0;--i){
        for(register int j=0;j&lt;=maxy;++j){
            putchar(pic[i][j]?pic[i][j]:&#039; &#039;);
        }
        putchar(&#039;\n&#039;);
    }
}

inline void paint(const int y,const int x,const int ch){
    int i=0;
    if(x&gt;maxx)maxx=x;
    if(y&gt;maxy)maxy=y;
    if(pic[x][y])return ;
    else {
        system(&quot;clear&quot;);//就是这里，windows要换成cls
        // while(i&lt;=10000)i++;
        print();
        pic[x][y]=ch;
    }
}

inline void draw(const int x,const int y){
    paint(x,y,&#039;+&#039;);
    paint(x+4,y,&#039;+&#039;);
    paint(x,y+3,&#039;+&#039;);
    paint(x+4,y+3,&#039;+&#039;);
    paint(x+6,y+2,&#039;+&#039;);
    paint(x+6,y+5,&#039;+&#039;);
    paint(x+2,y+5,&#039;+&#039;);
    paint(x+5,y+1,&#039;/&#039;);
    paint(x+5,y+4,&#039;/&#039;);
    paint(x+1,y+4,&#039;/&#039;);
    for(int i=1;i&lt;=3;++i){
        paint(x+i,y,&#039;-&#039;);
        paint(x+i,y+3,&#039;-&#039;);
        paint(x+i+2,y+5,&#039;-&#039;);
        paint(x+i,y+1,&#039; &#039;);
        paint(x+i,y+2,&#039; &#039;);
        paint(x+i+1,y+4,&#039; &#039;);
    }
    for(int i=1;i&lt;=2;++i){
        paint(x,y+i,&#039;|&#039;);
        paint(x+4,y+i,&#039;|&#039;);
        paint(x+6,y+2+i,&#039;|&#039;);
        paint(x+5,y+1+i,&#039; &#039;);
    }
}

void draw(int x,int y,int z){
    draw(2*(x-1)+4*(y-1),2*(x-1)+3*(z-1));
}

int main(){
    scanf(&quot;%d%d&quot;,&amp;n,&amp;m);
    for(int i=n;i;--i){
        for(int j=1;j&lt;=m;j++){
            scanf(&quot;%d&quot;,&amp;mana[i][j]);
            mana[i][0]=max(mana[i][0],mana[i][j]);
        }
    }
    for(int i=1;i&lt;=n;i++){
        for(int k=mana[i][0];k;k--){
            for(int j=m;j;--j){
                if(mana[i][j]&gt;=k)draw(i,j,k);
            }
        }
    }
    print();
    return 0;
}
