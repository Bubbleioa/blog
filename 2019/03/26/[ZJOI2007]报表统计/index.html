[ZJOI2007]报表统计

题目链接

题解

又是一道题一晚上系列(数据结构一生之敌)

调了很久的原因其实是写错了一个变量,其实还是很简单的

这里用的平衡树是Treap

你可以把原始数列看成n个链表,首尾相连的那种,但是我们要分段存
用vector就很舒服

然后对于第三个操作,其实只要每次进入平衡树的元素查找下前驱后继与它本身作差取最小值即可(这里的前驱后继是非严格的)

对于第二个操作,每次插入一个元素,总会创造两个差值关系并且减去一个差值关系
我们可以用两个小根堆分别维护产生的差值和删去的差值,每次询问时,不断弹出顶端相同元素后,第一个堆顶就是答案了

这种做法还是有点慢的(最慢1500ms,开了O2 790ms),如果当年省选不开O2的话有点悬

代码

#include&lt;bits/stdc++.h&gt;
using namespace std;
const int MAXN=600304,INF=0x7fffffff;
int pos[MAXN];//可以不用
int minn=INF;
vector&lt;int&gt;b[MAXN];

struct delhp{
    priority_queue&lt;int&gt;q1,q2;
    void insert(int kind,int x){
        if(kind==1)q1.push(-x);//小根堆
        else q2.push(-x);
    }
    int top(){
        while(q1.size()&amp;&amp;q2.size()){
            if(q1.top()!=q2.top())return -q1.top();
            q1.pop();q2.pop();
        }
    }
}q;

struct Treap{
    int l,r;
    int val,dat;
    int cnt,size;
}a[MAXN];
int tot,root;

int read(){
    int x=0,f=1;char ch=getchar();
    while(!isdigit(ch)){if(ch==&#039;-&#039;)f=-1;ch=getchar();}
    while(isdigit(ch)){x=x*10+ch-&#039;0&#039;;ch=getchar();}
    return x*f;
}

void update(int p){
    a[p].size=a[a[p].l].size+a[a[p].r].size+a[p].cnt;
}

int New(int x){
    a[++tot].val=x;
    a[tot].dat=rand();
    a[tot].size=a[tot].cnt=1;
    return tot;
}

void build(){
    New(-INF);New(INF);
    a[1].r=2;
    root=1;
    update(root);
}

void zig(int &amp;p){
    int q=a[p].l;
    a[p].l=a[q].r;a[q].r=p;p=q;
    update(a[p].r);update(p);
}

void zag(int &amp;p){
    int q=a[p].r;
    a[p].r=a[q].l;a[q].l=p;p=q;
    update(a[p].l);update(p);
}

void insert(int &amp;p,int val){
    if(p==0){
        p=New(val);
        return;
    }
    if(val==a[p].val){
        a[p].cnt++;
        update(p);
        return;
    }
    if(val&lt;a[p].val){
        insert(a[p].l,val);
        if(a[a[p].l].dat&gt;a[p].dat)zig(p);
    }
    else {
        insert(a[p].r,val);
        if(a[a[p].r].dat&gt;a[p].dat)zag(p);
    }
    update(p);
}

int GetPre(int val){
    int ans=1;
    int p=root;
    while(p){
        if(val==a[p].val){
            if(a[p].cnt&gt;1)return a[p].val;
            else if(a[p].l&gt;0){
                p=a[p].l;
                while(a[p].r&gt;0)p=a[p].r;
                ans=p;
            }
            break;
        }
        if(a[p].val&lt;val&amp;&amp;a[p].val&gt;a[ans].val)ans=p;
        p=val&lt;a[p].val?a[p].l:a[p].r;
    }
    return a[ans].val;
}

int GetNext(int val){
    int ans=2;
    int p=root;
    while(p){
        if(val==a[p].val){
            if(a[p].cnt&gt;1)return a[p].val;
            else if(a[p].r&gt;0){
                p=a[p].r;
                while(a[p].l&gt;0)p=a[p].l;
                ans=p;
            }
            break;
        }
        if(a[p].val&gt;val&amp;&amp;a[p].val&lt;a[ans].val)ans=p;
        p=val&lt;a[p].val?a[p].l:a[p].r;
    }
    return a[ans].val;
}

int main(){
    int n=read(),m=read();
    build();
    for(int i=1;i&lt;=n;i++){
        int x=read();
        pos[i]=x;
        b[i].push_back(x);
        if(i!=1)q.insert(1,abs(pos[i]-pos[i-1]));
        if(minn!=0){//维护操作3
            insert(root,pos[i]);
            int xx=GetNext(pos[i]);
            int yy=GetPre(pos[i]);
            if(xx!=INF&amp;&amp;xx!=-INF)minn=min(minn,xx-pos[i]);
            if(yy!=INF&amp;&amp;yy!=-INF)minn=min(minn,pos[i]-yy);
        }
    }
    for(int i=1;i&lt;=m;i++){
        char s[20];
        scanf(&quot;%s&quot;,s);
        int len=strlen(s);
        if(len==6){
            int x=read(),k=read();
            b[x].push_back(k);
            if(minn!=0)insert(root,k);
            q.insert(1,abs(k-*(b[x].end()-2)));//*号为解除引用,如果不加上则返回的是迭代器
            if(b[x+1].size()){//防止越界RE
                q.insert(1,abs(k-*(b[x+1].begin())));
                q.insert(2,abs(*(b[x].end()-2)-*(b[x+1].begin())));
            }
            if(minn!=0){
                int xx=GetNext(k);
                int yy=GetPre(k);
                if(xx!=INF&amp;&amp;xx!=-INF)minn=min(minn,xx-k);
                if(yy!=INF&amp;&amp;yy!=-INF)minn=min(minn,k-yy);
            }
        }
        else if(len==7){
            printf(&quot;%d\n&quot;,q.top());
        }
        else {
            printf(&quot;%d\n&quot;,minn);
        }
    }
    return 0;
}
