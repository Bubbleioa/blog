CF1155A Reverse a Substring

如果字符大小非严格单增，显然无解，如果出现前面字符大于后面的字符的情况，输出它们的位置即可。

#include&lt;bits/stdc++.h&gt;
using namespace std;
int n;string str;

int main(){
    cin&gt;&gt;n&gt;&gt;str;char last=str[0];int pos=0;
    for(int i=1;i&lt;str.length();i++){
        if(last==str[i]){pos=i;continue;}
        if(last&lt;str[i]){last=str[i];pos=i;}
        else {
            printf(&quot;YES\n%d %d&quot;,pos+1,i+1);
            return 0;
        }
    }
    printf(&quot;NO&quot;);
    return 0;
}

CF1155B Game with Telephone Numbers

一开始还以为是博弈论（我傻了）……结果一个简单的贪心就OK了

一轮游戏会将数字取到只剩11个，如果序列中前 $n-10$ 个元素中，8的数量大于其他的数量，那么显然先手会把其中不是8的取完，先手必胜。如果不大于，后手会把所有的8取完，先手必败。

#include&lt;bits/stdc++.h&gt;
using namespace std;
int n;int a[100005]; 

int main(){
    cin&gt;&gt;n;int cnt=0;
    for(int i=1;i&lt;=n;i++)scanf(&quot;%1d&quot;,&amp;a[i]);
    for(int i=1;i&lt;=n-10;i++){
         if(a[i]==8)cnt++;
         else cnt--;
    }
    if(cnt&gt;0)printf(&quot;YES\n&quot;);
    else
        printf(&quot;NO\n&quot;);
    return 0;
}

CF1155C Alarm Clocks Everywhere

我们要找到一个 x和d 使得 $x+kd = a_i$ k为正整数

如果我们的时间累积到 $a_i$ 了，那么我们其实就是要满足 $a_i+kd=a_{i+1}$ 即 $a_{i+1}-a_i=kd$

所以我们就是要用一个间隔来遍历到所有时间点，这个间隔显然是所有相邻时间间隔的gcd，记为d

然后看看第二行里面有没有d的约数，有的话就输出

#include&lt;bits/stdc++.h&gt;
using namespace std;
typedef long long ll;
const int MAXN=300003;
int n,m;
ll x,d,z;

ll gcd(ll a,ll b){
    return b?gcd(b,a%b):a;
}

int main(){
    scanf(&quot;%d%d&quot;,&amp;n,&amp;m);
    scanf(&quot;%lld&quot;,&amp;x);
    for (int i=1;i&lt;n;i++) {
        scanf(&quot;%lld&quot;, &amp;z);
        d=gcd(d,abs(z-x));
    }
    for(int i=1;i&lt;=m;i++){
        ll p;cin&gt;&gt;p;
        if(d%p==0){
            printf(&quot;YES\n%lld %d&quot;,x,i);
            return 0;
        }
    }
    printf(&quot;NO&quot;);
    return 0;
}

CF1155D Beautiful Array

其实就是最长连续子串的变形。

变量名 变量作用
$s$ $s_i$ 表示整个序列乘上 $x$ 后的前缀和
$L$ $L_i$ 表示 $i$ 位置往前的最大连续子串和
$R$ $R_i$ 表示 $i$ 位置往后的最大连续子串和

假如我们选择了 $i$ 到 $j$ ($j\le i$) 这一段乘上 $x$ 那么答案就为 $$s[i]-s[j-1]+R[i+1]+L[j-1]$$

我们可以枚举 $s[i]+R[i+1]$ ，用 $max(L[j-1]-s[j-1])$ 来更新答案

#include&lt;bits/stdc++.h&gt;
using namespace std;
typedef long long ll;
ll a[300005];
ll s[300005],L[300005],R[300005];

int main() {
    int n;ll x;
    cin&gt;&gt;n&gt;&gt;x;
    ll ans=0;
    for (int i=1; i&lt;=n; i++) {cin&gt;&gt;a[i];ans+=a[i];s[i]=x*a[i];s[i]+=s[i-1];}
    ans=max(ans,0LL);
    for (int i=1; i&lt;=n; i++) L[i]=max(L[i-1]+a[i],0LL);
    for (int i=n; i&gt;=1; i--) R[i]=max(R[i+1]+a[i],0LL);
    ll mx=-1LL&lt;&lt;62;
    for (int i=1; i&lt;=n; i++) {
        mx=max(mx,L[i-1]-s[i-1]);
        ans=max(ans,R[i+1]+s[i]+mx);
    }
    cout&lt;&lt;ans&lt;&lt;endl;
    return 0;
}
