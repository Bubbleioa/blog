#include&lt;bits/stdc++.h&gt;
using namespace std;
const int MAXN=10003;
typedef pair&lt;int,int&gt; P;
int Head[MAXN],to[MAXN&lt;&lt;1],Nt[MAXN&lt;&lt;1],tot;
int match[MAXN];
bool vis[MAXN];
int rmatch[MAXN];
int n,m;
P master[MAXN],dog[MAXN];

double dis(int x_1,int y_1,int x_2,int y_2){
    return sqrt((x_1-x_2)*(x_1-x_2)+((y_1-y_2)*(y_1-y_2)));
}

void add(int x,int y){
    Nt[++tot]=Head[x];
    to[tot]=y;
    Head[x]=tot;
}

int dfs(int x){
    for(int i=Head[x],y;i;i=Nt[i]){
        if(vis[y=to[i]])continue;
        vis[y]=1;
        if(!match[y]||dfs(match[y])){
            match[y]=x;return true;
        }
    }
    return false;
}

        }
    }
