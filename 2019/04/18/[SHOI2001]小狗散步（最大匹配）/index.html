[SHOI2001]小狗散步（最大匹配）

题目链接

题解

把人走的点当作左部节点，🐕走的当作右部
对于每个人走的节点，如果他与下一个节点的距离的两倍大于某个从当前节点走到右部节点+右部节点走到下一节点的距离，即可连边
输出方案的话，注意到如果右部节点被匹配，match[i]就是匹配它的左部节点，存起来输出即可

代码

#include&lt;bits/stdc++.h&gt;
using namespace std;
const int MAXN=10003;
typedef pair&lt;int,int&gt; P;
int Head[MAXN],to[MAXN&lt;&lt;1],Nt[MAXN&lt;&lt;1],tot;
int match[MAXN];
bool vis[MAXN];
int rmatch[MAXN];
int n,m;
P master[MAXN],dog[MAXN];

double dis(int x_1,int y_1,int x_2,int y_2){
    return sqrt((x_1-x_2)*(x_1-x_2)+((y_1-y_2)*(y_1-y_2)));
}

void add(int x,int y){
    Nt[++tot]=Head[x];
    to[tot]=y;
    Head[x]=tot;
}

int dfs(int x){
    for(int i=Head[x],y;i;i=Nt[i]){
        if(vis[y=to[i]])continue;
        vis[y]=1;
        if(!match[y]||dfs(match[y])){
            match[y]=x;return true;
        }
    }
    return false;
}

int main(){
    cin&gt;&gt;n&gt;&gt;m;
    for(int i=1;i&lt;=n;i++){
        int x,y;cin&gt;&gt;x&gt;&gt;y;
        master[i]=make_pair(x,y);
    }
    for(int i=1;i&lt;=m;i++){
        int x,y;cin&gt;&gt;x&gt;&gt;y;
        dog[i]=make_pair(x,y);
    }
    for(int i=1;i&lt;n;i++){
        for(int j=1;j&lt;=m;j++){
            if(dis(master[i].first,master[i].second,master[i+1].first,master[i+1].second)*2&gt;dis(master[i].first,master[i].second,dog[j].first,dog[j].second)+dis(dog[j].first,dog[j].second,master[i+1].first,master[i+1].second)){
                add(i,j+n);
            }
        }
    }
    int ans=0;
    for(int i=1;i&lt;n;i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i))ans++;
    }
    printf(&quot;%d\n&quot;,ans+n);
    for(int j=1;j&lt;=m;j++){
        if(match[j+n]){
            rmatch[match[j+n]]=j;
        }
    }
    for(int i=1;i&lt;=n;i++){
        printf(&quot;%d %d &quot;,master[i].first,master[i].second);
        if(i!=n&amp;&amp;rmatch[i]){
            printf(&quot;%d %d &quot;,dog[rmatch[i]].first,dog[rmatch[i]].second);
        }
    }
    return 0;
}
