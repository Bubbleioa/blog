[TJOI2017]DNA（字符串+哈希）

题目链接

题解

题目大意：给两个串，S，T，问S中有多少个子串与T的差异是在3个字符以内的。

这道题解法有很多，像什么SAM，FFT，NTT之类的，然而我只会哈希orz

二分找它们的最长公共前缀，最多能跳3次，没了……

设N为S的长度，M为T的长度
时间复杂度 $O((N-M)\times (logM))$

代码

#include&lt;bits/stdc++.h&gt;
using namespace std;
typedef unsigned long long ull;
const int P=131,MAXN=100005;
ull hash1[MAXN];
ull hash2[MAXN];
ull Q[MAXN];
int n,len1,len2;string str;

int num(char ch){
    if(ch==&#039;A&#039;)return 0;
    if(ch==&#039;T&#039;)return 1;
    if(ch==&#039;C&#039;)return 2;
    return 3;
}

ull ask(int type,int l,int r){
    if(type==1)return hash1[r]-hash1[l-1]*Q[r-l+1];
    else return hash2[r]-hash2[l-1]*Q[r-l+1];
}

int lcp(int x,int y,int r){
    int l=1,mid,t=0;
    while(l&lt;=r){
        mid=(l+r)&gt;&gt;1;
        if(ask(1,x,x+mid-1)==ask(2,y,y+mid-1)){
            t=mid;
            l=mid+1;
        }
        else r=mid-1;
    }
    return t;
}

bool check(int x){
    int r=x+len2-1,y=1;
    for(int i=0;i&lt;3;i++){
        int t=lcp(x,y,len2-y+1);
        x+=t+1;
        y+=t+1;
        if(y&gt;len2)return 1;
    }
    return ask(1,x,r)==ask(2,y,len2);
}

void solve(){
    int ans=0;
    for(int i=1;i&lt;=len1-len2+1;i++){
        if(check(i))ans++;
    }
    cout&lt;&lt;ans&lt;&lt;endl;
}

int main(){
    Q[0]=1;
    for(int i=1;i&lt;=100000;i++)Q[i]=Q[i-1]*P;
    cin&gt;&gt;n;
    while(n--){
        cin&gt;&gt;str;
        len1=str.length();
        for(int i=0;i&lt;len1;i++){
            i+=1;
            hash1[i]=hash1[i-1]*P+num(str[i-1]);
            i-=1;
        }
        cin&gt;&gt;str;
        len2=str.length();
        for(int i=0;i&lt;len2;i++){
            i+=1;
            hash2[i]=hash2[i-1]*P+num(str[i-1]);
            i-=1;
        }
        solve();
        memset(hash1,0,sizeof(hash1));
        memset(hash2,0,sizeof(hash2));
    }
    return 0;
}
